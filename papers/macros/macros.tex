    \ifbool{showcomments}{%
        \colorbox{#1}{\textcolor{#4}{\parbox{.8\linewidth}{#2: #3}}}%

%Definition Commands:
\newcounter{definition} % Counter for definitions
\renewcommand{\thedefinition}{\arabic{definition}} % Define the format of the counter

\newcommand{\define}[2]{%
    \par\vspace{\baselineskip}% Add some space before the definition
    \noindent\textbf{Definition \thedefinition}\ [#1]:\quad #2% Print the definition
    \stepcounter{definition}% Increment the counter
}


\newcounter{hypothesis} % Counter for hypotheses
\renewcommand{\thehypothesis}{\arabic{hypothesis}} % Define the format of the counter

\newcommand{\hyp}[1]{%
    \noindent\textbf{H\thehypothesis}:\quad\parbox[t]{\dimexpr\linewidth-2em\relax}{#1\\}% Print the Hypothesis
    \stepcounter{hypothesis}% Increment the counter
}
